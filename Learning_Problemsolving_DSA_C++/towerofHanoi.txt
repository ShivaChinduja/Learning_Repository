//Tower of Hanoi

#include <iostream>
using namespace std;

void towerofhanoi(int n , char s , char d ,char a){
    if(n==1){
        cout << "Move disk "<< n << " from " << s << " to "<< d<< endl;
    
        return;
    }
    
    towerofhanoi(n-1, s,a,d);
    
    cout << "Move disk "<< n << " from " << s << " to "<< d << endl;
    
    towerofhanoi(n-1,a,d,s);
}

int main() {
    int disk;
    cin >> disk;
    towerofhanoi(disk , 's' ,'d','a');
    

    return 0;
}
